            NOT = {
                controlled_by = ROOT
            }
        }

        days = 15
    }

            limit = {
                has_colonial_parent = ROOT

                days = 20
            }
        }
    }
    option = {
    #flee
        name = "capital_occupied.EVTOPT2"
        random_subject_country = {
            limit = {
                has_colonial_parent = ROOT
            }

        }
        every_subject_country = {
            limit = {
                has_colonial_parent = ROOT
                NOT = {tag = event_target:inheritor}
            }
            event_target:inheritor = {
                inherit = PREV
            }                     
        }    
    }
}
