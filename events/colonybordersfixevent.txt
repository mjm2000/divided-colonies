namespace = colonial_borders

# if colonial/former colonial country with colony vassals
# somehow gets land in their vassal's territory - give it to vassal



country_event = {
	id = colonial_borders.1
	title = "colonial_borders.EVTNAME1"
	desc = "colonial_borders.EVTDESC1"
	picture = BIG_BOOK_eventPicture

	trigger = {
		OR = {
			is_colonial_nation = yes
			is_former_colonial_nation = yes
		}
		NOT = { has_country_flag = keeping_acadia }
		capital_scope = {
			OR = {
				continent = north_america
				continent = south_america
			}
		}
		any_country = {
			vassal_of = ROOT
			capital_scope = {
				colonial_region = colonial_acadia
			}
		}
		any_province = {
			owned_by = ROOT
			colonial_region = colonial_acadia
		}
	}
	mean_time_to_happen = {
		days = 1
	}			
	option = {
		name = "colonial_borders.EVTOPT1" # cede the province
		colonial_acadia = {
			limit = {
				owned_by = ROOT
			}
			set_province_flag = transferring_acadia
		}
		every_country = {
			limit = {
				vassal_of = ROOT
				capital_scope = {
					colonial_region = colonial_acadia
				}
			}
			country_event = {
				id = colonial_borders.2
			}
		}
		ai_chance = {
			factor = 40
		}
	}
	option = {
		name = "colonial_borders.EVTOPT2" # keep the province
		set_country_flag = keeping_acadia
		ai_chance = {
			factor = 0
		}
	}
}

#second part of event
