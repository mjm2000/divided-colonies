namespace = colonial_border_cleanup_core

# if your colonial nations overgrow their boundaries

country_event = {
	id = colonial_border_cleanup_core.1
	title = "colonial_border_cleanup.EVTNAME1"
	desc = "colonial_border_cleanup.EVTDESC1"
	picture = BIG_BOOK_eventPicture

	is_triggered_only = yes

	trigger = {
		any_province = {
			colonial_region = colonial_acadia
			owner = {
				is_colonial_nation_of = ROOT
				capital_scope = {
					NOT = {
						colonial_region = colonial_acadia
					}
				}
			}
			is_core = owner
			NOT = {
				has_province_flag = allowed_acadia
			}
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = "colonial_border_cleanup.EVTOPT1" #fix the provinces
		every_province = {
			limit = {
				colonial_region = colonial_acadia
				owner = {
					is_colonial_nation_of = ROOT
					capital_scope = {
						NOT = {
							colonial_region = colonial_acadia
						}
					}
				}
				NOT = { has_province_flag = allowed_acadia }
			}
			cede_province = ROOT
			add_core = ROOT
		}
	}
	option = {
		name = "colonial_border_cleanup.EVTOPT2" #allow them to keep provinces
		every_province = {
			limit = {
				colonial_region = colonial_acadia
				owner = {
					is_colonial_nation_of = ROOT
					capital_scope = {
						NOT = {
							colonial_region = colonial_acadia
						}
					}
				}
			}
			set_province_flag = allowed_acadia
		}
		ai_chance = {
			factor = 0
		}
	}
}
