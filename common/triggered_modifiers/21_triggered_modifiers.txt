# countries that are former colonies that have other former colonies
# get free diplo slot for those vassals
# this is only within a certain region, AKA event vassals

# America

state_of_maine = {
	potential = {
		is_former_colonial_nation = yes
		capital_scope = {
			province_group = american_territory
		}	
	}

	trigger = {
		is_former_colonial_nation = yes
		any_country = {
			vassal_of = ROOT
			is_former_colonial_nation = yes
			capital_scope = {
				colonial_region = colonial_maine
			}
		}	
	}
	diplomatic_upkeep = 1
}
